PubNub Dev Lib

var pubnub = PUBNUB({
publish_key: 'pub-c-your-publish-key',
subscribe_key: 'sub-c-your-subscribe-key'
})

pubnub.subscribe({
channel: 'temperature-ds18b20',
message: function(m){console.log("Message received: " + JSON.stringify(m))}
});

pubnub.unsubscribe({
channel: 'temperature-ds18b20'
});
